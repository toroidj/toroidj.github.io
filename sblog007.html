ListFileName = PPx.Extract("%'tmp'¥result.txt")
SearchString = LCase(PPx.Extract("%""検索キーワード""%E"))
FolderName = PPx.Extract("%1")

Set ObjListFile = FSO.CreateTextFile(ListFileName,True)

ObjListFile.WriteLine ";ListFile"
SearchFolder FSO.GetFolder(FolderName)
ObjListFile.Close

Sub SearchFolder(ObjSrcFolder)
Set ObjSearchFolders = ObjSrcFolder.SubFolders

SearchComment ObjSrcFolder.Path,0 ' MultiByte
SearchComment ObjSrcFolder.Path,-1 ' UNICODE
For Each ObjFolder In ObjSearchFolders
' ObjListFile.WriteLine ObjFolder.Path
SearchFolder ObjFolder
Next
End Sub

Sub SearchComment(ObjSrcFile,code)
Dim FileName

FileName = ObjSrcFile & "¥00_index.txt"
If Not FSO.FileExists(FileName) Then Exit Sub
Set ObjFile = FSO.OpenTextFile(FileName,1,False,code)
Do Until ObjFile.AtEndOfStream
LineString = ObjFile.ReadLine
If InStr(LCase(LineString),SearchString) Then
ObjListFile.WriteLine ObjSrcFile & "¥" & Left(LineString,InStrRev(LineString,chr(9)))
End If
Loop
ObjFile.Close
End Sub
